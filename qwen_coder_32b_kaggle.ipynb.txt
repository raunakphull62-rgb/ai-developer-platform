    "from fastapi.middleware.cors import CORSMiddleware\n",
    "from pydantic import BaseModel\n",
    "from typing import List, Optional\n",
    "from pyngrok import ngrok, conf\n",
    "\n",
    "nest_asyncio.apply()\n",
    "conf.get_default().auth_token = NGROK_TOKEN\n",

    "app.add_middleware(\n",
    "    CORSMiddleware,\n",
    "    allow_origins=[\"*\"],\n",
    "    allow_methods=[\"*\"],\n",
    "    allow_headers=[\"*\"],\n",
    ")\n",
    "\n",
    "# ---- Data Models ----\n",
    "class Message(BaseModel):\n",
    "    role: str\n",
    "    content: str\n",
    "\n",
    "class ChatRequest(BaseModel):\n",
    "    model: Optional[str] = \"qwen-coder\"\n",
    "    messages: List[Message]\n",
    "    max_tokens: Optional[int] = 1024\n",
    "    temperature: Optional[float] = 0.2    # Low temp = more precise code\n",
    "    top_p: Optional[float] = 0.9\n",
    "\n",
    "# ---- Text Generator ----\n",
    "def generate(messages, max_tokens, temperature, top_p):\n",
    "    # Format using Qwen's chat template\n",
    "    chat_messages = []\n",
    "    for msg in messages:\n",
    "        chat_messages.append({\n",
    "            \"role\": msg.role,\n",
    "            \"content\": msg.content\n",
    "        })\n",
    "\n",
    "    # Apply Qwen2.5 chat template\n",
    "    text = tokenizer.apply_chat_template(\n",
    "        chat_messages,\n",
    "        tokenize=False,\n",
    "        add_generation_prompt=True\n",
    "    )\n",
    "\n",
    "    inputs = tokenizer([text], return_tensors=\"pt\").to(\"cuda:0\")\n",
    "\n",
    "    with torch.no_grad():\n",
    "        outputs = model.generate(\n",
    "            **inputs,\n",
    "            max_new_tokens=max_tokens,\n",
    "            temperature=temperature,\n",
    "            top_p=top_p,\n",
    "            do_sample=temperature > 0,\n",
    "            pad_token_id=tokenizer.eos_token_id,\n",
    "            eos_token_id=tokenizer.eos_token_id,\n",
    "        )\n",
    "\n",
    "    # Decode only the new tokens (not the input)\n",
    "    new_tokens = outputs[0][inputs.input_ids.shape[1]:]\n",
    "    return tokenizer.decode(new_tokens, skip_special_tokens=True).strip()\n",
    "\n",
    "# ---- Endpoints ----\n",

    "# ============================================================\n",
    "# CELL 7 (OPTIONAL): Upload Qwen Model TO Storj.io\n",
    "# Run ONCE after Cell 4 loads the model the first time.\n",
    "# Takes ~30-60 mins to upload but saves 18 mins every session!\n",
    "# ============================================================\n",
    "\n",
    "import os\n",
    "os.environ[\"PYDEVD_DISABLE_FILE_VALIDATION\"] = \"1\"\n",
    "\n",
    "import boto3\n",
    "from pathlib import Path\n",
    "\n",
    "# ---- YOUR STORJ CREDENTIALS ----\n",
    "STORJ_ACCESS_KEY = \"YOUR_STORJ_ACCESS_KEY\"\n",
    "STORJ_SECRET_KEY = \"YOUR_STORJ_SECRET_KEY\"\n",
    "STORJ_ENDPOINT   = \"https://gateway.storjshare.io\"\n",
    "BUCKET_NAME      = \"ai-models\"\n",
    "# --------------------------------\n",
    "\n",
    "s3 = boto3.client(\"s3\",\n",
    "    aws_access_key_id=STORJ_ACCESS_KEY,\n",
    "    aws_secret_access_key=STORJ_SECRET_KEY,\n",
    "    endpoint_url=STORJ_ENDPOINT\n",
    ")\n",
    "\n",
    "def upload_folder(local_folder, s3_prefix):\n",
    "    local_folder = Path(local_folder)\n",
    "    if not local_folder.exists():\n",
    "        print(f\"⚠️  Not found: {local_folder}\")\n",
    "        return\n",
    "    files = [f for f in local_folder.rglob(\"*\") if f.is_file()]\n",
    "    total_gb = sum(f.stat().st_size for f in files) / 1024**3\n",
    "    print(f\"Uploading {len(files)} files ({total_gb:.1f} GB) ...\")\n",
    "    for i, fp in enumerate(files):\n",
    "        key = f\"{s3_prefix}/{fp.relative_to(local_folder)}\".replace(\"\\\\\", \"/\")\n",
    "        print(f\"  [{i+1}/{len(files)}] {fp.name}\")\n",
    "        s3.upload_file(str(fp), BUCKET_NAME, key)\n",
